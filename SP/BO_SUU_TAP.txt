CREATE PROCEDURE sp_KiemTraThemBST
    @MABST NVARCHAR(50)
AS
BEGIN
    SET NOCOUNT ON;

    IF EXISTS (SELECT 1 FROM BO_SUU_TAP WHERE MABST = @MABST)
        RETURN 1;

    RETURN 0;
END


CREATE PROCEDURE sp_XoaBST
    @MABST NVARCHAR(50)
AS
BEGIN
    SET NOCOUNT ON;

    IF EXISTS (SELECT 1 FROM MUON WHERE MABST = @MABST AND MATPNT IS NOT NULL)
        RETURN 1;

    RETURN 0;
END
