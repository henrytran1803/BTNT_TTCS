CREATE PROCEDURE Them_CTL
    @MACTL nchar(8)
AS
BEGIN
    SET NOCOUNT ON;

    IF EXISTS (SELECT 1 FROM CUOC_TRIEN_LAM WHERE MACTL = @MACTL)
    BEGIN
        RETURN 1;
    END
    ELSE
    BEGIN
        RETURN 0;
    END
END


CREATE PROCEDURE Xoa_CTL
    @MACTL nchar(8)
AS
BEGIN
    SET NOCOUNT ON;

    IF EXISTS (SELECT 1 FROM CT_CTL WHERE MACTL = @MACTL)
    BEGIN
        RETURN 1;
    END
END
